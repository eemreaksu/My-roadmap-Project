# Clean Code — Pythonic yazım # ❌ Kötü def f(x, y, z): r = [] for i in x: if i[y] > z: r.append(i) return r # ✅ İyi — type hints, anlamlı isimler def yasa_gore_filtrele( kullanicilar: list[dict], alan: str, esik: int ) -> list[dict]: """Belirtilen alana göre kullanıcıları filtreler.""" return [u for u in kullanicilar if u[alan] > esik] # Algoritmalar — sorting örneği def quicksort(dizi: list) -> list: if len(dizi) <= 1: return dizi pivot = dizi[len(dizi) // 2] sol = [x for x in dizi if x < pivot] orta = [x for x in dizi if x == pivot] sag = [x for x in dizi if x > pivot] return quicksort(sol) + orta + quicksort(sag)
